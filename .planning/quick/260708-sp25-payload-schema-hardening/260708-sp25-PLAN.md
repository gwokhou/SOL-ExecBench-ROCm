# Payload Schema Hardening Implementation Plan

> **Legacy execution note:** This imported plan was originally written for the Superpowers workflow. Use the surrounding GSD quick-task artifact and current repository instructions for any future execution; the original checkbox steps remain for historical traceability.

**Goal:** Reduce scattered raw `dict` / `.get()` payload handling by moving JSON compatibility logic to explicit parser boundaries and typed schema objects.

**Architecture:** Keep permissive parsing at external artifact boundaries, but normalize raw JSON into Pydantic models, dataclasses, or typed payload access helpers before core report/scoring logic runs. Treat report builders and sidecar readers as boundary adapters; core aggregation code should consume typed records or normalized dictionaries with narrow helper APIs.

**Tech Stack:** Python 3.12, Pydantic v2, pytest, Ruff, existing `src/sol_execbench/core/data/json_utils.py`, existing uncommitted `src/sol_execbench/core/data/path_access.py` if accepted.

## Boundary Classification

Use these categories during implementation:

- **Allowed raw boundary modules:** file readers, historical artifact loaders, report sidecar parsers, migration compatibility readers, and tests that directly inspect JSON examples.
- **Migration targets:** modules where raw JSON is immediately converted to typed domain records and then used internally without repeated `isinstance(value, dict)` checks.
- **Do not migrate mechanically:** environment variables, counters, graph `attributes` dictionaries, and intentionally flexible metadata maps.

```python
"""Helpers for nested payload access."""

from __future__ import annotations

import math
from collections.abc import Mapping
from typing import Any

from glom import PathAccessError, glom


def path_get(payload: object, path: str, *, default: Any = None) -> Any:
    """Return a nested payload value or a default when the path is absent."""
    return glom(payload, path, default=default)


def path_require(payload: object, path: str, *, source: str = "payload") -> Any:
    """Return a nested payload value, raising ValueError when it is absent."""
    try:
        return glom(payload, path)
    except PathAccessError as exc:
        raise ValueError(f"{source} missing required field: {path}") from exc


def path_dict(
    payload: object,
    path: str,
    *,
    default: dict[str, Any] | None = None,
) -> dict[str, Any]:
    value = path_get(payload, path)
    if isinstance(value, Mapping):
        return dict(value)
    return {} if default is None else dict(default)


def path_list(payload: object, path: str) -> list[Any]:
    value = path_get(payload, path)
    return list(value) if isinstance(value, list) else []


def path_str_or_none(payload: object, path: str) -> str | None:
    value = path_get(payload, path)
    return value if isinstance(value, str) else None


def path_int(payload: object, path: str, *, default: int = 0) -> int:
    value = path_get(payload, path)
    if isinstance(value, bool):
        return default
    return value if isinstance(value, int) else default


def path_float_or_none(payload: object, path: str) -> float | None:
    value = path_get(payload, path)
    if isinstance(value, bool) or not isinstance(value, int | float):
        return None
    number = float(value)
    return number if math.isfinite(number) else None
```

```python
from __future__ import annotations

import ast
from pathlib import Path


SOURCE_ROOT = Path(__file__).resolve().parents[2] / "src"
PACKAGE_ROOT = SOURCE_ROOT / "sol_execbench"


RAW_PAYLOAD_ALLOWLIST = {
    "sol_execbench.cli.commands",
    "sol_execbench.cli.environment",
    "sol_execbench.cli.problem_io",
    "sol_execbench.core.baseline_export",
    "sol_execbench.core.claim_upgrade",
    "sol_execbench.core.consistency",
    "sol_execbench.core.dataset.migration",
    "sol_execbench.core.dataset.paper_denominator.stages",
    "sol_execbench.core.dataset.profiler_timing_coverage",
    "sol_execbench.core.evaluation_stability",
    "sol_execbench.core.matrix_diff",
    "sol_execbench.core.scoring.amd_bound_sanity_helpers",
    "sol_execbench.core.scoring.amd_score_sidecar_parsing",
    "sol_execbench.core.scoring.baseline_artifact",
    "sol_execbench.core.scoring.solar_derivation_parse_root",
}


def _module_name(path: Path) -> str:
    module = ".".join(path.relative_to(SOURCE_ROOT).with_suffix("").parts)
    return module.removesuffix(".__init__")


def _has_raw_payload_shape_checks(path: Path) -> bool:
    tree = ast.parse(path.read_text(encoding="utf-8"))
    for node in ast.walk(tree):
        if not isinstance(node, ast.Call):
            continue
        if isinstance(node.func, ast.Name) and node.func.id == "isinstance":
            if len(node.args) == 2 and isinstance(node.args[1], ast.Name):
                if node.args[1].id == "dict":
                    return True
        if isinstance(node.func, ast.Attribute) and node.func.attr == "get":
            return True
    return False


def test_raw_payload_shape_checks_stay_in_allowlisted_boundary_modules() -> None:
    modules_with_shape_checks = {
        _module_name(path)
        for path in PACKAGE_ROOT.rglob("*.py")
        if _has_raw_payload_shape_checks(path)
    }

    unexpected = sorted(modules_with_shape_checks - RAW_PAYLOAD_ALLOWLIST)

    assert unexpected == []
```

## Execution Notes

- Do not convert every `.get()` mechanically. Environment maps, graph node attributes, counters, and intentionally extensible metadata dictionaries can remain dictionary-based.
- Every parser should preserve existing compatibility behavior unless a test explicitly updates the contract.
- Raw JSON compatibility should live in named parser helpers such as `_profiler_timing_summary_from_payload()` or `minimal_amd_sol_bound_v2_from_payload()`.
- Business logic should receive typed records, Pydantic models, or normalized helper-return dictionaries rather than arbitrary raw JSON payloads.
