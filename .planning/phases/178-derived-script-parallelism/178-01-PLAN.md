def _partition_targets_by_index(
    targets: list[Path],
    max_workers: int,
) -> list[list[Path]]:
    """Pre-partition targets for exclusive worker ownership."""
    if not targets:
        return []
    chunk_size = (len(targets) + max_workers - 1) // max_workers
    chunks = []
    for i in range(0, len(targets), chunk_size):
        chunks.append(targets[i : i + chunk_size])
    return chunks

def _process_target_chunk(
    chunk: list[Path],
    *,
    args: argparse.Namespace,
    benchmark_dir: Path,
    status_lock: threading.Lock,
) -> list[ProblemStatus]:
    """Process a chunk of targets serially within a worker thread."""
    chunk_results = []
    for target in chunk:
        # Process target, append status with lock
        with status_lock:
            append_status(args.status_jsonl, status)
        chunk_results.append(status)
    return chunk_results

# Main execution pattern:
with ThreadPoolExecutor(max_workers=max_workers) as executor:
    futures = {executor.submit(_process_target_chunk, chunk, ...): chunk for chunk in chunks}
    for future in as_completed(futures.keys()):
        results.extend(future.result())
